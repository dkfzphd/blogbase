library(gganimate)
library(ggthemes)

theme_set(theme_economist_white())

x <- seq(0,15,by=0.1)
y <- sin(x)+0.8*x+0.1*x^2
df1 <- tibble(x,y)
plot1 <- ggplot(df1,aes(x,y))+
  geom_line(size=1,color='#e63946')+
  geom_line(aes(x,1.9*x),size=1,color='black')+
  geom_point(color='#03045e', alpha=0.5, size=3)+
  labs(x="Time",
       y="Ability")
plot1

animate1 <- plot1 +
              transition_reveal(x)
animate1
